<!DOCTYPE html>
<html>
 <head>
     <script>
        var ctx;
        var x=-100;

        function startcanvas()
        {
            var c = document.getElementById("canvasid");
            ctx = c.getContext("2d");

            redraw();
        }

       function drawfish()
       {
            ctx.beginPath();
            ctx.moveTo(46,52);
            ctx.bezierCurveTo(100, 100, 75, 10, 180, 5);
            ctx.quadraticCurveTo(196, 120, 40, 60);
            ctx.lineTo(15,70);
            ctx.quadraticCurveTo(50,50,44,25);
            ctx.lineWidth = 3;
            ctx.strokeStyle = 'black';
            ctx.closePath();

            ctx.fill();
            ctx.stroke();
       }

        function redraw()
        {
            ctx.clearRect(0,0,200,200);

            ctx.save();
            
            gradient=ctx.createLinearGradient(0,30,0,150);
            gradient.addColorStop(0,"#da7");  
            gradient.addColorStop(0.5,"#00f");
            gradient.addColorStop(1,"#fff");
            ctx.fillStyle=gradient;
            drawfish();
            
            ctx.translate(0,75);
            
            gradient=ctx.createRadialGradient(140,100,20,180,100,200);
            gradient.addColorStop(0,"#8ed6ff");
            gradient.addColorStop(0.5,"#f00");
            gradient.addColorStop(1,"#004CB3");
            ctx.fillStyle=gradient;
            drawfish();

            ctx.restore();

            window.requestAnimationFrame(redraw);
        }
     </script>
 </head>
 <body onload="startcanvas();">
     <canvas id="canvasid" width="200" height="200" style="border:1px solid red;width:400px;height:400px;image-rendering: pixelated;">
        
     </canvas>
 </body>
</html>
